    <script>
        document.getElementById('contactForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const nombre = document.getElementById('nombre').value.trim();
            const email = document.getElementById('email').value.trim();
            const asunto = document.getElementById('asunto').value.trim();
            const categoria = document.getElementById('categoria').value;
            const mensaje = document.getElementById('mensaje').value.trim();
            
            // Validaciones
            if(!nombre || nombre.length < 3) {
                alert('Por favor ingresa un nombre válido (mínimo 3 caracteres)');
                return;
            }
            
            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            if(!email || !emailRegex.test(email)) {
                alert('Por favor ingresa un email válido');
                return;
            }
            
            if(!asunto || asunto.length < 5) {
                alert('Por favor ingresa un asunto válido (mínimo 5 caracteres)');
                return;
            }
            
            if(!categoria) {
                alert('Por favor selecciona una categoría');
                return;
            }
            
            if(!mensaje || mensaje.length < 10) {
                alert('Por favor escribe un mensaje más largo (mínimo 10 caracteres)');
                return;
            }
            
            // Mostrar éxito
            alert('✓ ¡Mensaje enviado con éxito!\nNos pondremos en contacto pronto.');
            this.reset();
            document.getElementById('nombre').focus();
        });
    </script>
